   var a = 12, b = 3;
   var add, sub, mul, div, modulus;

   sub = a - b; //subtract 3 from 12
   mul = a * b; //Multiplying both
   div = a / b; //dividing 12 by 3 (number of times)
   modulus = a % b; //calculation the remainder

   document.write("Addition of " +a +' and ' +b +" is = " + add + "<br />");
   document.write("Subtraction of " +a +' and ' +b +" is = " + sub + "<br />");
   document.write("Multiplication of " +a +' and ' +b +" is = " + mul + "<br />");
   document.write("Division of " +a +' and ' +b +" is = " + div + "<br />");
   document.write("Modulus of " +a +' and ' +b +" is = " + modulus + "<br />");
